\section{The $q$-binomial theorem: statement and proof}
\label{sec:qbin-proof}

\subsection{Statement}

Define the \emph{finite} $q$-Pochhammer symbol
\[
(a; q)_n = \prod_{k=0}^{n-1} (1 - a q^k), \qquad (a; q)_0 = 1,
\]
the \emph{$q$-factorial} $[n]_q! = (q;q)_n / (1-q)^n$, and the \emph{Gaussian binomial coefficient}
\[
\binom{n}{k}_q = \frac{(q; q)_n}{(q; q)_k \, (q; q)_{n-k}} \qquad (0 \leq k \leq n).
\]
By convention $\binom{n}{k}_q = 0$ for $k < 0$ or $k > n$. These are polynomials in $q$ with integer coefficients (this is part of what needs to be proved).

\begin{theorem}[Finite $q$-binomial theorem]\label{thm:qbin-finite}\lean{qSeries.qBinom_finite_thm}\leanok\uses{def:qBinom, def:qPochhammer}
In $\mathbb{Z}[q, z]$, for every $n \geq 0$,
\[
\prod_{k=0}^{n-1} (1 + z q^k) = \sum_{k=0}^n q^{\binom{k}{2}} \binom{n}{k}_q z^k.
\]
\end{theorem}

\begin{theorem}[Infinite $q$-binomial / Cauchy identity]\label{thm:qbin-infinite}\lean{qSeries.qBinom_infinite_thm}\leanok\uses{def:qPochhammer, def:qPochhammerInf}
For $|q| < 1$ and any $a, z \in \mathbb{C}$ with $|z| < 1$,
\[
\sum_{n=0}^\infty \frac{(a; q)_n}{(q; q)_n} z^n = \frac{(a z; q)_\infty}{(z; q)_\infty}.
\]
\end{theorem}

The two Euler identities follow from Theorem~\ref{thm:qbin-infinite} by specializing $a = 0$ (gives $1/(z;q)_\infty = \sum z^n / (q;q)_n$) and by sending $a \to \infty$ along $a = -y/q^N$ with $N \to \infty$, then renaming (gives $(-z;q)_\infty = \sum q^{\binom{n}{2}} z^n / (q;q)_n$).

\subsection{Proof of the finite $q$-binomial theorem}

This proof is purely algebraic (no analysis) and is the version best suited for Lean. It uses only induction on $n$ and the $q$-Pascal identity.

\begin{lemma}[$q$-Pascal]\label{lem:qpascal}\lean{qSeries.qBinom_succ_succ}\leanok\uses{def:qBinom}
For all $n, k \in \mathbb{N}$,
\[
  \binom{n+1}{k+1}_q \;=\; \binom{n}{k+1}_q + q^{n - k}\,\binom{n}{k}_q.
\]
(Equivalently, with $k \geq 1$:
$\binom{n+1}{k}_q = \binom{n}{k}_q + q^{n - k + 1}\binom{n}{k-1}_q$.
A symmetric form $\binom{n+1}{k+1}_q = q^{k+1}\binom{n}{k+1}_q + \binom{n}{k}_q$
also holds; we use the first.)
\end{lemma}

\begin{proof}\leanok
This is the recursive defining equation of $\binom{n}{k}_q$ in
Definition~\ref{def:qBinom}, hence \texttt{rfl} in Lean. Equivalently, by
direct computation from the quotient form
$\binom{n}{k}_q = (q;q)_n / \bigl((q;q)_k(q;q)_{n-k}\bigr)$
(valid when $k \leq n$ and the denominators are nonzero):
$\binom{n}{k}_q + q^{n-k+1}\binom{n}{k-1}_q
  = \binom{n}{k-1}_q \cdot \tfrac{(1 - q^{n-k+1}) + q^{n-k+1}(1 - q^k)}{1 - q^k}
  = \binom{n}{k-1}_q \cdot \tfrac{1 - q^{n+1}}{1 - q^k}
  = \binom{n+1}{k}_q$,
recovering the off-by-one form.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:qbin-finite}]\proves{thm:qbin-finite}\leanok\uses{lem:qpascal, lem:qBinom-eq-zero-of-lt, def:qBinom}
Induction on $n$, following the Lean proof
\texttt{qSeries.qBinom\_finite\_thm} in \texttt{Series/qSeries.lean}.
The case $n = 0$ gives $1 = 1$.

For the inductive step, set
$P_n(z) := \prod_{k=0}^{n-1}(1 + z q^k)$ and
$Q_n(z) := \sum_{k=0}^{n} q^{\binom{k}{2}}\binom{n}{k}_q z^k$,
so the theorem reads $P_n = Q_n$. By \texttt{Finset.prod\_range\_succ}
and the inductive hypothesis $P_n = Q_n$,
\[
  P_{n+1}(z) \;=\; P_n(z)\,(1 + z q^n) \;=\; Q_n(z)\,(1 + z q^n),
\]
so it suffices to show $Q_{n+1}(z) = Q_n(z)\,(1 + z q^n)$. We decompose
$Q_{n+1}(z) = \sum_{k=0}^{n+1} q^{\binom{k}{2}}\binom{n+1}{k}_q z^k$ in
four explicit steps (mirroring \texttt{stepB}--\texttt{stepD} of the
Lean proof).

\emph{Step A (peel off $k = 0$ via \texttt{Finset.sum\_range\_succ'}).}
Using $\binom{n+1}{0}_q = 1$,
\[
  Q_{n+1}(z) \;=\; 1 \;+\; \sum_{k=0}^{n} q^{\binom{k+1}{2}}\,\binom{n+1}{k+1}_q\,z^{k+1}.
\]

\emph{Step B ($q$-Pascal split).} Apply Lemma~\ref{lem:qpascal}
inside the sum and split via \texttt{Finset.sum\_add\_distrib}:
\[
  \sum_{k=0}^{n} q^{\binom{k+1}{2}}\binom{n+1}{k+1}_q z^{k+1}
    \;=\; \underbrace{\sum_{k=0}^{n} q^{\binom{k+1}{2}}\binom{n}{k+1}_q z^{k+1}}_{S_1}
       \;+\; \underbrace{\sum_{k=0}^{n} q^{\binom{k+1}{2}+n-k}\binom{n}{k}_q z^{k+1}}_{S_2}.
\]

\emph{Step C ($S_2 = z q^n\, Q_n(z)$).} The identity
$\binom{k+1}{2} = \binom{k}{2} + k$ (helper
\texttt{choose\_two\_succ}) and $k + (n - k) = n$ (valid because
$k \leq n$, by \texttt{Nat.sub\_add\_cancel}) give
$q^{\binom{k+1}{2}+n-k}\, z^{k+1} = z\, q^n \cdot q^{\binom{k}{2}}\, z^k$, hence
\[
  S_2 \;=\; z\, q^n \sum_{k=0}^{n} q^{\binom{k}{2}}\binom{n}{k}_q z^k \;=\; z\, q^n\, Q_n(z).
\]

\emph{Step D ($1 + S_1 = Q_n(z)$).} Reindex $j = k+1$ to identify
$S_1 = \sum_{j=1}^{n+1} q^{\binom{j}{2}}\binom{n}{j}_q z^j$, then
adjoin the $j = 0$ term $1$ (using \texttt{Finset.sum\_range\_succ'}
in reverse):
\[
  1 + S_1 \;=\; \sum_{j=0}^{n+1} q^{\binom{j}{2}}\binom{n}{j}_q z^j.
\]
The $j = n+1$ term vanishes by Lemma~\ref{lem:qBinom-eq-zero-of-lt}
($\binom{n}{n+1}_q = 0$), reducing this to $Q_n(z)$.

Combining the four steps,
$Q_{n+1}(z) = (1 + S_1) + S_2 = Q_n(z) + z q^n Q_n(z) = Q_n(z)\,(1 + z q^n)$.
The final algebraic identity is closed in Lean by
\texttt{linear\_combination}. \qedhere
\end{proof}

\subsection{Passing to the infinite case}

\begin{proof}[Proof of Theorem~\ref{thm:qbin-infinite}]\leanok
\proves{thm:qbin-infinite}
\uses{def:qPochhammer, def:qPochhammerInf, lem:multipliable-one-sub-smul-qpow}
We follow Heine's classical functional-equation argument, which is the route
realised in Lean (\texttt{qSeries.qBinom\_infinite\_thm}). Define
\[
  c_n \;:=\; \frac{(a;q)_n}{(q;q)_n}, \qquad
  F(z) \;:=\; \sum_{n \ge 0} c_n\, z^n, \qquad
  G(z) \;:=\; \frac{(a z;q)_\infty}{(z;q)_\infty}.
\]
The argument has three pieces.

\emph{(i) Coefficient recurrence.} From $(a;q)_{n+1} = (a;q)_n(1 - a q^n)$ and
$(q;q)_{n+1} = (q;q)_n(1 - q^{n+1})$,
\[
  c_{n+1}\,(1 - q^{n+1}) \;=\; c_n\,(1 - a q^n).
\]
(Note that $|q| < 1$ forces $1 - q^{n+1} \neq 0$, so $c_n$ is well-defined.)

\emph{(ii) Functional equation $(1-z)\,H(z) = (1-az)\,H(qz)$ for $H \in \{F, G\}$.}
For $F$: expand both sides as power series; the coefficient identity in (i)
shows the series of $(1-z) F(z)$ and $(1-az) F(qz)$ agree term-by-term. The
sum $F(z)$ converges absolutely on $\|z\| < 1$ because the coefficient
sequence $c_n$ is bounded (its numerator $\|(a;q)_n\|$ converges by
Lemma~\ref{lem:multipliable-one-sub-smul-qpow} and its denominator $\|(q;q)_n\|$
is eventually bounded below by a positive constant).
For $G$: from the telescoping identity $(z;q)_\infty = (1-z)\,(zq;q)_\infty$
(an instance of $\prod_{k\ge 0} f_k = f_0 \cdot \prod_{k\ge 1} f_k$) applied
to both $(z;q)_\infty$ and $(az;q)_\infty$.

\emph{(iii) Iteration and passage to the limit.} Iterating the functional
equation $n$ times on the open unit disc gives
\[
  F(z) \cdot (z;q)_n \;=\; F(q^n z) \cdot (a z;q)_n
\]
for every $n \ge 0$. As $n \to \infty$:
\begin{itemize}
  \item $(z;q)_n \to (z;q)_\infty$ and $(az;q)_n \to (az;q)_\infty$ by
    Lemma~\ref{lem:multipliable-one-sub-smul-qpow} (the partial products of a
    multipliable family converge to its \texttt{tprod});
  \item $F(q^n z) \to F(0) = c_0 = 1$: this is Tannery's theorem
    (\texttt{tendsto\_tsum\_of\_dominated\_convergence}) applied summand-by-summand,
    with the dominating bound $\|c_k (q^n z)^k\| \le C\,\|z\|^k$ (summable in
    $k$, uniformly in $n$).
\end{itemize}
Therefore $F(z) \cdot (z;q)_\infty = (a z;q)_\infty$. Since $\|z\| < 1$
implies $(z;q)_\infty \neq 0$ (no factor $1 - z q^k$ can be zero, by the
norm estimate; combined with multipliability this gives non-vanishing of the
\texttt{tprod}), we may divide to conclude $F(z) = G(z)$. \qedhere
\end{proof}

\paragraph{Remark (alternative route).} A different classical route passes
$n \to \infty$ in a Cauchy-type \emph{finite} analog
\[
  \sum_{k=0}^n \binom{n}{k}_q \frac{(a;q)_k}{(q;q)_k} z^k
  \;\xrightarrow{n \to \infty}\;
  \sum_{k=0}^\infty \frac{(a;q)_k}{(q;q)_k} z^k,
\]
using $(q;q)_k / (q;q)_n \to 1/(q^{k+1};q)_\infty$ and dominated convergence.
The Heine route used above avoids the bookkeeping of a separate finite identity
and is the one matched by the Lean formalisation.

\paragraph{Avoiding the infinite version entirely.}
For Jacobi's triple product via Zhu's semi-finite proof, you do not actually need Theorem~\ref{thm:qbin-infinite}. You only need the \emph{first Euler identity}
\[
(-z; q)_\infty = \sum_{n=0}^\infty \frac{q^{\binom{n}{2}}}{(q; q)_n} z^n,
\]
which is the $n \to \infty$ limit of Theorem~\ref{thm:qbin-finite} after rewriting $\prod_{k=0}^{n-1}(1 + zq^k)$ as $(-z; q)_n$ and dividing by suitable factors. This limit is gentler than the full Cauchy identity.

\section{The $q$-Pochhammer symbol and Gaussian binomial coefficient (formalized)}
\label{sec:qPochhammer-formalized}

This section records the foundational definitions and lemmas that have been
formalized in Lean for the $q$-series approach to Jacobi's triple product.
The corresponding source file is \texttt{Series/qSeries.lean} in the
\texttt{Series} Lean library, under the namespace \texttt{qSeries}.
Throughout this section $R$ is a commutative ring and $a, q \in R$.

\subsection{The $q$-Pochhammer symbol}

\begin{definition}[Finite $q$-Pochhammer symbol]\label{def:qPochhammer}\lean{qSeries.qPochhammer}\leanok
For $n \in \mathbb{N}$, the finite $q$-Pochhammer symbol is
\[
  (a; q)_n \;:=\; \prod_{k = 0}^{n - 1} \bigl(1 - a\, q^k\bigr).
\]
In Lean, this is implemented as \texttt{$\prod$ k $\in$ Finset.range n, (1 - a * q\^{}k)},
specialising to $(a; q)_0 = 1$ (the empty product).
\end{definition}

\begin{lemma}[Base case]\label{lem:qPochhammer-zero}\lean{qSeries.qPochhammer_zero}\leanok\uses{def:qPochhammer}
$(a; q)_0 = 1$.
\end{lemma}

\begin{proof}\leanok
The empty product is $1$ (\texttt{Finset.prod\_range\_zero}).
\end{proof}

\begin{lemma}[Recurrence]\label{lem:qPochhammer-succ}\lean{qSeries.qPochhammer_succ}\leanok\uses{def:qPochhammer}
For all $n \in \mathbb{N}$,
\[
  (a; q)_{n+1} \;=\; (a; q)_n \cdot \bigl(1 - a\, q^n\bigr).
\]
\end{lemma}

\begin{proof}\leanok
Immediate from \texttt{Finset.prod\_range\_succ}.
\end{proof}

\subsection{The Gaussian binomial coefficient}

\begin{definition}[Gaussian binomial coefficient]\label{def:qBinom}\lean{qSeries.qBinom}\leanok
The Gaussian binomial coefficient $\binom{n}{k}_q \in R$ is defined by
recursion on $n$ and $k$:
\[
\begin{aligned}
  \binom{0}{0}_q &= 1,
  & \binom{0}{k+1}_q &= 0, \\
  \binom{n+1}{0}_q &= 1,
  & \binom{n+1}{k+1}_q &= \binom{n}{k+1}_q + q^{n-k}\,\binom{n}{k}_q.
\end{aligned}
\]
This is the $q$-Pascal recurrence in the form stated in
Lemma~\ref{lem:qpascal} (with the index shift $k \mapsto k+1$). By
construction $\binom{n}{k}_q$ is a polynomial in $q$ with integer
coefficients, defined for every pair $(n, k) \in \mathbb{N}^2$ (no
division, no roots-of-unity exclusion).
\end{definition}

\begin{lemma}[Vanishing above the diagonal]\label{lem:qBinom-eq-zero-of-lt}\lean{qSeries.qBinom_eq_zero_of_lt}\leanok\uses{def:qBinom}
If $n < k$, then $\binom{n}{k}_q = 0$.
\end{lemma}

\begin{proof}\leanok
Induction on $n$ and $k$ following the four-case recursive definition.
The non-trivial case is $(n+1, k+1)$ with $n < k$: the recurrence gives
$\binom{n+1}{k+1}_q = \binom{n}{k+1}_q + q^{n-k}\binom{n}{k}_q$, and both
inductive hypotheses (at $(n, k+1)$ and $(n, k)$, available because
$n < k + 1$ and $n < k$ respectively) yield $0$.
\end{proof}

\begin{lemma}[Diagonal]\label{lem:qBinom-self}\lean{qSeries.qBinom_self}\leanok\uses{def:qBinom, lem:qBinom-eq-zero-of-lt}
$\binom{n}{n}_q = 1$ for every $n \in \mathbb{N}$.
\end{lemma}

\begin{proof}\leanok
Induction on $n$. The case $n = 0$ is by definition. For the step,
\[
  \binom{n+1}{n+1}_q
    = \binom{n}{n+1}_q + q^{n-n}\,\binom{n}{n}_q
    = 0 + 1 \cdot 1 = 1,
\]
using Lemma~\ref{lem:qBinom-eq-zero-of-lt} for the first term and the
inductive hypothesis for the second.
\end{proof}

\begin{lemma}[Closed form for $\binom{n}{k}_q$]\label{lem:qBinom-closed-form}\lean{qSeries.qBinom_mul_qPochhammer_mul_qPochhammer}\leanok\uses{def:qBinom, def:qPochhammer, lem:qPochhammer-succ, lem:qBinom-self, lem:qBinom-eq-zero-of-lt}
For all $k \leq n$,
\[
  \binom{n}{k}_q \cdot (q; q)_k \cdot (q; q)_{n - k} \;=\; (q; q)_n.
\]
\end{lemma}

\begin{proof}\leanok
Induction on $n$.

\emph{Base case $n = 0$.} Forces $k = 0$ and the identity is $1 \cdot 1 \cdot 1 = 1$.

\emph{Boundary $k = 0$.} Reduces to $1 \cdot 1 \cdot (q; q)_n = (q; q)_n$.

\emph{Boundary $k = n + 1$.} By Lemma~\ref{lem:qBinom-self} we have
$\binom{n+1}{n+1}_q = 1$, and $(q; q)_0 = 1$, so both sides equal
$(q; q)_{n+1}$.

\emph{Inductive step $k + 1 \leq n$.} Apply the $q$-Pascal recurrence
\[
  \binom{n+1}{k+1}_q = \binom{n}{k+1}_q + q^{n-k}\,\binom{n}{k}_q
\]
and the factorisations
\[
  (q; q)_{n - k} \;=\; (q; q)_{n - (k+1)} \cdot \bigl(1 - q\,q^{n-(k+1)}\bigr),
  \qquad
  (q; q)_{k + 1} \;=\; (q; q)_k \cdot \bigl(1 - q\,q^k\bigr),
\]
together with the power identities $q \cdot q^{n-(k+1)} = q^{n-k}$ and
$q^{n-k} \cdot (q \cdot q^k) = q \cdot q^n$, which use $k \leq n$ via
\texttt{Nat.sub\_add\_cancel}. The two inductive hypotheses at $(n, k+1)$
and $(n, k)$ then combine into the required identity, dispatched in Lean
by \texttt{linear\_combination}.
\end{proof}

\subsection{The finite $q$-binomial theorem (step 5)}\label{subsec:qbin-finite-formalized}

Theorem~\ref{thm:qbin-finite} is formalised as
\texttt{qSeries.qBinom\_finite\_thm} in \texttt{QSeries/FiniteBinomial.lean}
(in \texttt{Qseries\_Aristotle\_new}).
% Old reference: \texttt{Series/qSeries.lean} (monolithic version, superseded).
The Lean proof follows exactly the four-step strategy spelled out in
the proof attached to Theorem~\ref{thm:qbin-finite} above:
peel off $k = 0$, apply the $q$-Pascal recurrence
(Lemma~\ref{lem:qpascal}) inside the sum, simplify the second piece

\subsection{The infinite $q$-Pochhammer symbol and Cauchy identity (steps 6, 8)}\label{subsec:qbin-infinite-formalized}

Working in $\mathbb{C}$ for the analytic part of the development.

\begin{definition}[Infinite $q$-Pochhammer symbol]\label{def:qPochhammerInf}\lean{qSeries.qPochhammerInf}\leanok
\[
  (a; q)_\infty \;:=\; \prod_{k=0}^{\infty}\bigl(1 - a\, q^k\bigr) \;\in\; \mathbb{C},
\]
defined unconditionally via \texttt{tprod}. The value is mathematically
meaningful (and the product is convergent) under the hypothesis $\|q\| < 1$
provided by Lemma~\ref{lem:multipliable-one-sub-smul-qpow}.
\end{definition}

\begin{lemma}[Multipliability]\label{lem:multipliable-one-sub-smul-qpow}\lean{qSeries.multipliable_one_sub_smul_qpow}\leanok\uses{def:qPochhammerInf}
For $a, q \in \mathbb{C}$ with $\|q\| < 1$, the family
$k \mapsto 1 - a\, q^k$ is multipliable; in particular the partial
products converge to $(a; q)_\infty$.
\end{lemma}

\begin{proof}\leanok
Reduce multipliability to summability of $\sum_k \|a q^k\|$ via Mathlib's
\texttt{multipliable\_one\_add\_of\_summable} (which works in any complete
normed ring), then bound $\|a q^k\| = \|a\|\, \|q\|^k$ and conclude by the
geometric series $\sum_k \|q\|^k < \infty$
(\texttt{summable\_geometric\_of\_lt\_one}).
\end{proof}

\begin{theorem}[Infinite $q$-binomial / Cauchy identity, restated for the formal development]\label{thm:qbin-infinite-formalized}\lean{qSeries.qBinom_infinite_thm}\leanok\uses{def:qPochhammer, def:qPochhammerInf, lem:multipliable-one-sub-smul-qpow}
For $a, z, q \in \mathbb{C}$ with $\|q\| < 1$ and $\|z\| < 1$,
\[
  \sum_{n=0}^{\infty} \frac{(a; q)_n}{(q; q)_n}\, z^n
    = \frac{(a z; q)_\infty}{(z; q)_\infty}.
\]
(Equivalent to Theorem~\ref{thm:qbin-infinite}.)
The Lean theorem is \texttt{qSeries.qBinom\_infinite\_thm} in
\texttt{QSeries/CauchyIdentity.lean} (in \texttt{Qseries\_Aristotle\_new}).
% Old reference: \texttt{Series/qSeries.lean} (monolithic version, superseded).
The proof realised in Lean follows Heine's
functional-equation argument spelled out in the proof of
Theorem~\ref{thm:qbin-infinite} above, organised under
\texttt{section CauchyIdentity}: non-vanishing of $(z;q)_\infty$
(\texttt{qPochhammerInf\_z\_q\_ne\_zero}), the telescoping recursion
$(z;q)_\infty = (1-z)\,(zq;q)_\infty$ (\texttt{qPochhammerInf\_recursion}),
the coefficient recurrence $c_{n+1}(1-q^{n+1}) = c_n(1 - a q^n)$
(\texttt{cauchyCoeff\_succ\_mul}), the functional equations for $F$ and $G$
(\texttt{cauchy\_functional\_eq\_F}, \texttt{cauchy\_functional\_eq\_G}),
iteration on the disc (\texttt{iterated\_functional\_eq\_disc}), and the
Tannery-based limit $F(q^n z) \to 1$ (\texttt{tendsto\_F\_qpow}). The
alternative route via a finite Cauchy-type analog of
Theorem~\ref{thm:qbin-finite} is not formalised.
\end{theorem}

\section{Implemented proof strategy}
\label{sec:implemented}

This section records the proof strategy that was \emph{actually} carried out in
the Lean formalisation (produced with the Aristotle prover). It does not follow
any one of the three textbook routes of Sections~\ref{sec:hermite},
\ref{sec:qbin}, \ref{sec:complex} verbatim. It is closest in spirit to
Proof~2 (the Euler / $q$-binomial route, Section~\ref{sec:qbin}), but the
execution is neither Andrews' nor Zhu's: instead of Zhu's semi-finite
truncation or a complex-analytic continuation, the formalisation

\begin{enumerate}
  \item expands $(q;q)_\infty(-z;q)_\infty$ and $(-q/z;q)_\infty$ by the
    \emph{second} Euler identity and forms their full Cauchy product on
    $\mathbb{N}\times\mathbb{N}$;
  \item evaluates every diagonal coefficient through the classical
    \emph{Durfee rectangle identity} $S_k(q) = 1/(q;q)_\infty$, here proved
    \emph{analytically} by a contraction recurrence on the differences
    $S_k - S_{k+1}$ rather than by the usual Young-diagram bijection;
  \item removes the convergence (annulus) restriction by proving an
    \emph{extended} second Euler identity valid for \emph{all} $z\in\mathbb{C}$,
    so the Cauchy-product argument runs directly on the punctured disc; and
  \item derives Euler's pentagonal number theorem from the triple product by the
    substitution $q\mapsto q^3$, $z\mapsto -q$.
\end{enumerate}

\paragraph{Source files.} The development lives in the Lean library
\texttt{Qseries\_Aristotle\_new}, split into the modules
\texttt{QSeries/Defs.lean}, \texttt{QSeries/FiniteBinomial.lean},
\texttt{QSeries/InfPochhammer.lean}, \texttt{QSeries/CauchyIdentity.lean},
\texttt{QSeries/EulerIdentities.lean}, \texttt{QSeries/JTP\_KeyIdentity.lean},
\texttt{QSeries/JTP\_Core.lean}, \texttt{QSeries/JTP\_Helpers.lean},
\texttt{QSeries/JacobiTripleProduct.lean},
\texttt{QSeries/PentagonalNumber.lean}, and
\texttt{QSeries/JTP\_Analytic.lean} (which extends the identity to all
$z \neq 0$), all under the namespace \texttt{qSeries}. An independent
purely algebraic proof of the triple product in the formal power series ring
$R[\![ X]\!]$ lives in \texttt{QSeries/FPS.lean},
\texttt{QSeries/FPS\_Euler.lean}, and \texttt{QSeries/FPS\_Algebra.lean}
under the namespace \texttt{qSeries.FPS}; this is documented in
Section~\ref{sec:fps}. The $q$-Pochhammer / $q$-binomial foundations
(Section~\ref{sec:qPochhammer-formalized}) are shared between both
developments.

\subsection{The two Euler identities}

Both Euler $q$-exponential identities are obtained as specialisations of the
Cauchy identity (Theorem~\ref{thm:qbin-infinite-formalized}) and the finite
$q$-binomial theorem (Theorem~\ref{thm:qbin-finite}).

\begin{lemma}[First Euler identity]\label{lem:euler-first}\lean{qSeries.euler_first_identity}\leanok\uses{thm:qbin-infinite-formalized, def:qPochhammerInf}
For $\|q\| < 1$ and $\|z\| < 1$,
\[
  \sum_{n=0}^{\infty} \frac{z^n}{(q;q)_n} = \frac{1}{(z;q)_\infty}.
\]
\end{lemma}

\begin{proof}\leanok
Specialise the Cauchy identity
(Theorem~\ref{thm:qbin-infinite-formalized}) at $a = 0$, using
$(0;q)_n = 1$ and $(0\cdot z;q)_\infty = 1$.
\end{proof}

\begin{lemma}[Second Euler identity]\label{lem:euler-second}\lean{qSeries.euler_second_identity}\leanok\uses{thm:qbin-finite, def:qPochhammer, def:qPochhammerInf, lem:qBinom-closed-form}
For $\|q\| < 1$ and $\|z\| < 1$,
\[
  \sum_{n=0}^{\infty} \frac{q^{\binom{n}{2}} z^n}{(q;q)_n} = (-z;q)_\infty.
\]
\end{lemma}

\begin{proof}\leanok
Take $N \to \infty$ in the finite $q$-binomial theorem
(Theorem~\ref{thm:qbin-finite}), rewriting
$\prod_{k=0}^{N-1}(1 + z q^k) \to (-z;q)_\infty$ by
Lemma~\ref{lem:multipliable-one-sub-smul-qpow} and replacing
$\binom{N}{k}_q$ by $(q;q)_N / \bigl((q;q)_k (q;q)_{N-k}\bigr)$
(Lemma~\ref{lem:qBinom-closed-form}). The exchange of limit and sum is
\texttt{tendsto\_tsum\_of\_dominated\_convergence}, with the dominating bound
coming from $(q;q)_N / (q;q)_{N-k} \to 1$ being bounded uniformly in $N$.
Summability of the target series is
\texttt{qSeries.euler\_second\_summable}.
\end{proof}

\subsection{Expanding the product as a single series}

The two factors $(q;q)_\infty(-z;q)_\infty$ are folded into one series using a
telescoping identity for the infinite $q$-Pochhammer symbol.

\begin{lemma}[Telescoping]\label{lem:qPochhammerInf-telescope}\lean{qSeries.qPochhammerInf_eq_mul}\leanok\uses{def:qPochhammer, def:qPochhammerInf}
For $\|q\| < 1$ and every $n$,
\[
  (q;q)_\infty \;=\; (q;q)_n \cdot (q^{\,n+1};q)_\infty.
\]
\end{lemma}

\begin{proof}\leanok
Induction on $n$, using $(q;q)_{n+1} = (q;q)_n(1 - q^{n+1})$
(Lemma~\ref{lem:qPochhammer-succ}) and the recursion
$(z;q)_\infty = (1-z)(zq;q)_\infty$
(\texttt{qSeries.qPochhammerInf\_recursion}).
\end{proof}

\begin{lemma}[Product as a series]\label{lem:qPochhammerInf-prod-hassum}\lean{qSeries.qPochhammerInf_prod_hasSum}\leanok\uses{lem:euler-second, lem:qPochhammerInf-telescope}
For $\|q\| < 1$ and $\|z\| < 1$,
\[
  \sum_{n=0}^{\infty} q^{\binom{n}{2}} z^n\,(q^{\,n+1};q)_\infty
    = (q;q)_\infty\,(-z;q)_\infty.
\]
\end{lemma}

\begin{proof}\leanok
Multiply the second Euler identity (Lemma~\ref{lem:euler-second}) by the
constant $(q;q)_\infty$ and rewrite $(q;q)_\infty/(q;q)_n = (q^{n+1};q)_\infty$
via Lemma~\ref{lem:qPochhammerInf-telescope}.
\end{proof}


\subsection{The key identity \texorpdfstring{$S_k = 1/(q;q)_\infty$}{Sk = 1/(q;q)inf}}

This is the core of the formalisation, replacing both the
coefficient-recurrence bookkeeping of Proof~1 and the complex analysis of
Proof~3. After the Cauchy product is rearranged by diagonals, each diagonal
coefficient is exactly one of the sums
\[
  S_k(q) \;:=\; \sum_{m=0}^{\infty}
    \frac{q^{\,m(m+k)}}{(q;q)_m\,(q;q)_{m+k}},
\]
and all of them collapse to $1/(q;q)_\infty$. This is the classical
\emph{Durfee rectangle identity}: combinatorially, $m\times(m+k)$ is the Durfee
rectangle of a partition, contributing $q^{m(m+k)}$, with the regions to its
right and below contributing $1/(q;q)_m$ and $1/(q;q)_{m+k}$; the $k=0$ case is
the classical Durfee square identity. (See Andrews, \emph{The Theory of
Partitions}, 1976, and Gessel, \emph{Some generalized Durfee square identities},
Discrete Math.\ \textbf{50} (1984).) What is specific to this formalisation is
the proof: rather than the Young-diagram bijection, the identity is established
\emph{analytically} through the contraction recurrence below, which is
convenient to formalise in Lean.

\begin{definition}[The key sum]\label{def:S-sum}\lean{qSeries.S_sum}\leanok\uses{def:qPochhammer}
For $\|q\| < 1$ and $k \in \mathbb{N}$,
$S_k(q) := \sum_{m \ge 0} q^{m(m+k)} / \bigl((q;q)_m\,(q;q)_{m+k}\bigr)$.
\end{definition}

\begin{lemma}[Summability of $S_k$]\label{lem:summable-S}\lean{qSeries.summable_S_summand}\leanok\uses{def:S-sum}
For $\|q\| < 1$, the summand of $S_k$ is summable.
\end{lemma}

\begin{proof}\leanok
The denominators $(q;q)_m(q;q)_{m+k}$ are bounded below in norm by a positive
constant (the partial products of a convergent infinite product with nonzero
limit), so the summand is dominated by $C\,\|q\|^{m^2}$, which is summable.
\end{proof}

\begin{lemma}[Limit of $S_k$]\label{lem:S-sum-tendsto}\lean{qSeries.S_sum_tendsto}\leanok\uses{def:S-sum, def:qPochhammerInf}
For $\|q\| < 1$, $S_k(q) \to 1/(q;q)_\infty$ as $k \to \infty$.
\end{lemma}

\begin{proof}\leanok
The $m = 0$ term is $1/(q;q)_k \to 1/(q;q)_\infty$
(\texttt{tendsto\_qPochhammer} and non-vanishing of $(q;q)_\infty$). The tail
$m \ge 1$ tends to $0$ by dominated convergence: each term
$q^{m(m+k)}/\bigl((q;q)_m(q;q)_{m+k}\bigr) \to 0$ as $k \to \infty$, dominated
by $\|q\|^{m^2}/C$.
\end{proof}

\begin{lemma}[Recurrence for $S_k$]\label{lem:S-sum-recurrence}\lean{qSeries.S_sum_recurrence}\leanok\uses{def:S-sum, lem:qPochhammer-succ}
For $\|q\| < 1$ and every $k$,
\[
  S_k - S_{k+1} \;=\; q^{\,k+1}\,\bigl(S_{k+2} - S_{k+1}\bigr).
\]
\end{lemma}

\begin{proof}\leanok
Combine the two series term-by-term, using $(q;q)_{m+k+1} = (q;q)_{m+k}(1 - q^{m+k+1})$
to write $S_k - S_{k+1}$ as a single series, then split the numerator
$1 - q^m - q^{m+k+1}$ and reindex the $(1 - q^m)$ part by $m \mapsto m+1$. The
two pieces reassemble into $q^{k+1}(S_{k+2} - S_{k+1})$.
\end{proof}

\begin{theorem}[Key identity]\label{thm:S-sum-eq}\lean{qSeries.S_sum_eq}\leanok\uses{lem:S-sum-recurrence, lem:S-sum-tendsto}
For $\|q\| < 1$ and every $k$, $\;S_k(q) = 1/(q;q)_\infty$.
\end{theorem}

\begin{proof}\leanok\proves{thm:S-sum-eq}
Write $D_k = S_k - S_{k+1}$. Iterating Lemma~\ref{lem:S-sum-recurrence}
$n$ times gives $\|D_k\| \le \|q\|^{\,n(2k+n+1)/2}\,\|D_{k+n}\|$. Since
$D_{k+n} \to 0$ (Lemma~\ref{lem:S-sum-tendsto}) and $\|q\|^{\,\cdots} \to 0$,
we get $D_k = 0$ for all $k$. Hence $S_k = S_0$ is constant in $k$, and its
common value equals $\lim_{k} S_k = 1/(q;q)_\infty$.
\end{proof}

\begin{lemma}[Non-negative diagonal coefficient]\label{lem:cauchy-coeff-nonneg}\lean{qSeries.cauchy_coeff_nonneg}\leanok\uses{thm:S-sum-eq}
For $\|q\| < 1$ and $k \ge 0$,
\[
  \sum_{m=0}^{\infty}
    q^{\binom{m+k}{2}}\,(q^{\,m+k+1};q)_\infty \cdot
    \frac{q^{\binom{m}{2}}\, q^{m}}{(q;q)_m} = q^{\binom{k}{2}}.
\]
\end{lemma}

\begin{proof}\leanok
Rewrite $(q^{m+k+1};q)_\infty = (q;q)_\infty/(q;q)_{m+k}$
(Lemma~\ref{lem:qPochhammerInf-telescope}) and use the arithmetic identity
$\binom{m+k}{2} + \binom{m}{2} + m = \binom{k}{2} + m(m+k)$ to reduce the sum to
$q^{\binom{k}{2}}\,(q;q)_\infty \cdot S_k(q)$. Apply
Theorem~\ref{thm:S-sum-eq}.
\end{proof}

\begin{lemma}[Negative diagonal coefficient]\label{lem:cauchy-coeff-neg}\lean{qSeries.cauchy_coeff_neg}\leanok\uses{thm:S-sum-eq}
For $\|q\| < 1$ and $l \ge 0$,
\[
  \sum_{n=0}^{\infty}
    q^{\binom{n}{2}}\,(q^{\,n+1};q)_\infty \cdot
    \frac{q^{\binom{n+l+1}{2}}\, q^{\,n+l+1}}{(q;q)_{n+l+1}}
    = q^{\binom{l+2}{2}}.
\]
\end{lemma}

\begin{proof}\leanok
As in Lemma~\ref{lem:cauchy-coeff-nonneg}, using the companion arithmetic
identity $\binom{n}{2} + \binom{n+l+1}{2} + (n+l+1) = \binom{l+2}{2} + n(n+l+1)$
and the key identity $S_{l+1} = 1/(q;q)_\infty$ (Theorem~\ref{thm:S-sum-eq}).
\end{proof}

\subsection{Extended second Euler identity (all \texorpdfstring{$z$}{z})}

To run the Cauchy product on the full punctured disc rather than only the
annulus, the second Euler identity is extended to all $z\in\mathbb{C}$.

\begin{lemma}[Series recursion]\label{lem:euler-second-recursion}\lean{qSeries.euler_second_series_recursion}\leanok\uses{def:qPochhammer, lem:qPochhammer-succ}
For $\|q\| < 1$, the series $S(z) := \sum_n q^{\binom{n}{2}} z^n / (q;q)_n$
satisfies $S(z) = (1+z)\,S(qz)$.
\end{lemma}

\begin{proof}\leanok
A term-by-term algebraic identity: the difference of the $z$- and $qz$-series
equals $z\,S(qz)$, using $(q;q)_{n} = (q;q)_{n-1}(1 - q^{n})$. Summability for
all $z$ is \texttt{qSeries.euler\_second\_summable\_all}, proved by the ratio
test (the ratio $\|q\|^n\|z\|/\|1 - q^{n+1}\| \to 0$).
\end{proof}

\begin{theorem}[Extended second Euler identity]\label{thm:euler-second-all}\lean{qSeries.euler_second_identity_all}\leanok\uses{lem:euler-second, lem:euler-second-recursion}
For $\|q\| < 1$ and \emph{every} $z\in\mathbb{C}$,
\[
  \sum_{n=0}^{\infty} \frac{q^{\binom{n}{2}} z^n}{(q;q)_n}
    = (-z;q)_\infty.
\]
\end{theorem}

\begin{proof}\leanok\proves{thm:euler-second-all}
Given any $z$, choose $N$ with $\|z q^N\| < 1$. Iterating
Lemma~\ref{lem:euler-second-recursion} $N$ times gives
$S(z) = \prod_{k=0}^{N-1}(1 + z q^k)\cdot S(z q^N)$, and
$S(z q^N) = (-z q^N;q)_\infty$ by the (small-argument) second Euler identity
(Lemma~\ref{lem:euler-second}). The product
$\prod_{k=0}^{N-1}(1 + z q^k)\cdot(-zq^N;q)_\infty = (-z;q)_\infty$ by repeated
use of \texttt{qSeries.qPochhammerInf\_recursion}.
\end{proof}

\begin{lemma}[Extended Euler identity at $q/z$]\label{lem:euler-second-qoz-all}\lean{qSeries.euler_second_at_qoz_all}\leanok\uses{thm:euler-second-all}
For $\|q\| < 1$ and $z \neq 0$ (no annulus condition),
\[
  \sum_{m=0}^{\infty} \frac{q^{\binom{m}{2}}\, q^{m}\, z^{-m}}{(q;q)_m}
    = (-q/z;q)_\infty.
\]
\end{lemma}

\begin{proof}\leanok
Apply Theorem~\ref{thm:euler-second-all} at the point $q/z$ (valid for all
$q/z$, hence all $z \neq 0$).
\end{proof}

\subsection{The triple product and bilateral series}

\begin{definition}[Triple product and bilateral series]\label{def:jacobiProd}\lean{qSeries.jacobiProd, qSeries.jacobiBilateral}\leanok\uses{def:qPochhammerInf}
$\mathrm{jacobiProd}(q,z) := (q;q)_\infty\,(-z;q)_\infty\,(-q/z;q)_\infty$, and
the bilateral series $\mathrm{jacobiBilateral}(q,z) :=
\sum_{k\ge0} z^k q^{\binom{k}{2}} + \sum_{m\ge0} z^{-(m+1)} q^{\binom{m+2}{2}}$
(the non-negative and negative halves of $\sum_{k\in\mathbb{Z}} z^k q^{k(k-1)/2}$).
\end{definition}

\subsection{The Jacobi triple product}

\begin{theorem}[Jacobi triple product]\label{thm:jtp-full}\lean{qSeries.jacobiTripleProduct}\leanok\uses{lem:qPochhammerInf-prod-hassum, lem:euler-second-qoz-all, lem:cauchy-coeff-nonneg, lem:cauchy-coeff-neg, def:jacobiProd}
For $\|q\| < 1$, $\|z\| < 1$, $z \neq 0$, $\;\mathrm{jacobiProd}(q,z) =
\mathrm{jacobiBilateral}(q,z)$; that is,
\[
  (q;q)_\infty\,(-z;q)_\infty\,(-q/z;q)_\infty
    = \sum_{k\ge0} z^k q^{\binom{k}{2}} + \sum_{m\ge0} z^{-(m+1)} q^{\binom{m+2}{2}}.
\]
\end{theorem}

\begin{proof}\leanok\proves{thm:jtp-full}
Expand $(q;q)_\infty(-z;q)_\infty$ as a series
(Lemma~\ref{lem:qPochhammerInf-prod-hassum}) and $(-q/z;q)_\infty$ via the
\emph{extended} second Euler identity (Lemma~\ref{lem:euler-second-qoz-all}),
which holds for every $z \neq 0$ with no lower bound on $\|z\|$. Their Cauchy
product over $\mathbb{N}\times\mathbb{N}$ is summable
(\texttt{Summable.mul\_norm}), so by \texttt{Summable.tsum\_prod} the double sum
may be reordered. Partition $\mathbb{N}\times\mathbb{N}$ into the diagonals
$n \ge m$ (index $k = n - m \ge 0$) and $n < m$ (index $l = m - n - 1 \ge 0$).
The $k$-diagonal sum is $z^k q^{\binom{k}{2}}$ by
Lemma~\ref{lem:cauchy-coeff-nonneg}, and the $l$-diagonal sum is
$z^{-(l+1)} q^{\binom{l+2}{2}}$ by Lemma~\ref{lem:cauchy-coeff-neg}. Since the
extended identity carries no annulus hypothesis, the argument runs directly on
the full punctured disc $0 < \|z\| < 1$, with no separate continuation step.
\end{proof}

\paragraph{Remark (divergence from the outline).} Compared with
Section~\ref{sec:qbin}: the formalisation uses a \emph{genuine Cauchy product}
of two second-Euler expansions, not Zhu's semi-finite truncation; the diagonal
coefficients are evaluated through the classical Durfee rectangle identity
(Theorem~\ref{thm:S-sum-eq}), but via its analytic contraction-recurrence proof
rather than the usual Young-diagram bijection; and the argument is carried out
directly on the whole punctured disc by strengthening the Euler identity to all
$z$ (Theorem~\ref{thm:euler-second-all}), so it needs neither an annulus
restriction with a separate extension step, the Liouville-type complex analysis
of Proof~3, nor the coefficient bookkeeping of Proof~1. (Using Durfee rectangles
to reach the Jacobi triple product is itself known; see W.\ Chu,
\emph{Durfee rectangles and the Jacobi triple product identity}, Acta Math.\
Sinica \textbf{9} (1993), 24--26.)

\subsection{Euler's pentagonal number theorem}

\begin{definition}[Generalized pentagonal number]\label{def:pentagonal}\lean{qSeries.pentagonal}\leanok
$\omega(k) := \bigl(k(3k-1)/2\bigr)$ for $k \in \mathbb{Z}$ (the product
$k(3k-1)$ is always even), realised in Lean as a natural number.
\end{definition}

\begin{theorem}[Euler's pentagonal number theorem]\label{thm:pentagonal-impl}\lean{qSeries.eulerPentagonalNumber}\leanok\uses{thm:jtp-full, def:pentagonal}
For $\|q\| < 1$,
\[
  (q;q)_\infty
    = \sum_{k\ge0} (-1)^k q^{\omega(k)}
    + \sum_{k\ge0} (-1)^{k+1} q^{\omega(-(k+1))}
    \;=\; 1 - q - q^2 + q^5 + q^7 - q^{12} - \cdots.
\]
\end{theorem}

\begin{proof}\leanok\proves{thm:pentagonal-impl}
Apply the Jacobi triple product (Theorem~\ref{thm:jtp-full}) with
$q \mapsto q^3$ and $z \mapsto -q$; the hypotheses reduce to $\|{-}q\| < 1$ and
$-q \neq 0$, both immediate for $0 < \|q\| < 1$ (the case $q = 0$ is trivial).
The left-hand side $(q^3;q^3)_\infty(q;q^3)_\infty(q^2;q^3)_\infty$ is regrouped
into $(q;q)_\infty$ by splitting the product $\prod_{n\ge1}(1 - q^n)$ into the
three residue classes $\{3k+1\},\{3k+2\},\{3k+3\}$ modulo $3$. Matching the
right-hand exponents $k^2$ (with sign $(-1)^k$) against
$\omega(\pm k)$ gives the pentagonal series.
\end{proof}

\subsection{Analytic extension to all \texorpdfstring{$z \neq 0$}{z ≠ 0}}
\label{subsec:analytic-extension}

Theorem~\ref{thm:jtp-full} was proved under the restriction $\|z\| < 1$.
The module \texttt{QSeries/JTP\_Analytic.lean} removes this restriction using
locally uniform convergence of both sides on $\{z \neq 0\}$ and the functional
equation $f(qz) = f(z)/z$ satisfied by both
$\mathrm{jacobiProd}(q,z)$ and $\mathrm{jacobiBilateral}(q,z)$.

\begin{theorem}[Summability of bilateral series for all $z$]\label{thm:summable-nonneg-all}\lean{qSeries.summable_jacobi_nonneg_all}\leanok\uses{def:jacobiProd}
For $\|q\| < 1$ and \emph{every} $z \in \mathbb{C}$, the non-negative part
$\sum_{k \ge 0} z^k\, q^{\binom{k}{2}}$ is summable.
\end{theorem}

\begin{proof}\leanok
The ratio of consecutive terms is $|z|\,|q|^n \to 0$, so the ratio test gives
summability; the Weierstrass $M$-test with $M_k = R^k \|q\|^{\binom{k}{2}}$
(summable by a further ratio test) gives uniform convergence on
$|z| \le R$.
\end{proof}

\begin{theorem}[Locally uniform convergence of the product side]\label{thm:jacobi-prod-luc}\lean{qSeries.jacobiProd_tendstoLocallyUniformlyOn}\leanok\uses{def:jacobiProd}
For $\|q\| < 1$, the partial products
$\prod_{k < N}(1 - q^{2k})(1 + zq^{2k-1})(1 + q^{2k-1}/z)$ converge
\emph{locally uniformly} on $\{z \neq 0\}$ to $\mathrm{jacobiProd}(q,z)$.
\end{theorem}

\begin{proof}\leanok
On $\overline{B}(z_0, \|z_0\|/2)$ for any $z_0 \neq 0$: the factor
$\prod(1 + (-z)q^k)$ is handled by the uniform convergence of
$\prod(1 + a_k)$ on compacta (Weierstrass $M$-test with
$\sum \|a_k\| < \infty$), and the factor $\prod(1 + (-q/z)q^k)$ uses the
bound $\|-q/z\| \le 2\|q\|/\|z_0\|$ on that ball. Products of uniformly
convergent bounded sequences converge uniformly.
\end{proof}

\begin{theorem}[Locally uniform convergence of the series side]\label{thm:jacobi-series-luc}\lean{qSeries.jacobiBilateral_tendstoLocallyUniformlyOn}\leanok\uses{def:jacobiProd}
For $\|q\| < 1$, the bilateral partial sums
$\sum_{|k| < N} z^k\, q^{\binom{k}{2}}$ converge \emph{locally uniformly}
on $\{z \neq 0\}$ to $\mathrm{jacobiBilateral}(q,z)$.
\end{theorem}

\begin{proof}\leanok
The non-negative half is locally uniformly convergent everywhere by
Theorem~\ref{thm:summable-nonneg-all} and the $M$-test. The negative half
$\sum_{m \ge 0} z^{-(m+1)} q^{\binom{m+2}{2}}$ is uniformly convergent on
$\{\|z^{-1}\| \le R'\}$; near any $z_0 \neq 0$ the ball
$B(z_0, \|z_0\|/2)$ satisfies $\|z^{-1}\| \le 2/\|z_0\|$.
\end{proof}

\begin{theorem}[Analytic Jacobi triple product]\label{thm:jtp-analytic}\lean{qSeries.jacobiTripleProduct_analytic}\leanok\uses{thm:jtp-full, def:jacobiProd, thm:summable-nonneg-all, thm:jacobi-prod-luc, thm:jacobi-series-luc}
For $\|q\| < 1$ and every $z \neq 0$,
\[
  (q;q)_\infty\,(-z;q)_\infty\,(-q/z;q)_\infty
    = \sum_{k \ge 0} z^k\, q^{\binom{k}{2}}
    + \sum_{m \ge 0} z^{-(m+1)}\, q^{\binom{m+2}{2}}.
\]
\end{theorem}

\begin{proof}\leanok\proves{thm:jtp-analytic}
Both sides are locally uniformly convergent on $\{z \neq 0\}$
(Theorems~\ref{thm:jacobi-prod-luc} and~\ref{thm:jacobi-series-luc}).
Both satisfy the functional equation $H(qz) = H(z)/z$ for all $z \neq 0$.
For any fixed $z \neq 0$, choose $N$ large enough so that $\|q^N z\| < 1$.
Then Theorem~\ref{thm:jtp-full} (valid at $q^N z$) together with the
iterated functional equation for both sides yields equality at $z$.
\end{proof}


\section{Formal power series reformulation (algebraic proof)}
\label{sec:fps}

This section records the second, independent proof of the Jacobi triple product
that was formalised in \texttt{Qseries\_Aristotle\_new/QSeries/FPS.lean},
\texttt{QSeries/FPS\_Euler.lean}, and \texttt{QSeries/FPS\_Algebra.lean} under the namespace
\texttt{qSeries.FPS}. Unlike the analytic proof of Section~\ref{sec:implemented},
this proof is \emph{entirely algebraic}: no norms, no $\|q\| < 1$ hypothesis, and
no topological analysis beyond the $X$-adic (pi) topology on $R[\![ X ]\!]$.

\paragraph{Key idea.}
The variable $q$ becomes the formal power series indeterminate $X \in R[\![ X]\!]$,
while $z$ lives in the coefficient ring
$A = \texttt{LaurentPolynomial}\,\mathbb{C}$ (with $z = T(1)$,
$z^{-1} = T(-1)$). The infinite $q$-Pochhammer symbol $(a; X)_\infty$
is a well-defined element of $R[\![ X]\!]$ because the factors
$(1 - a X^k)$ converge to $1$ in the pi topology: each factor affects only
finitely many coefficients of any fixed degree.

\subsection{FPS q-Pochhammer symbols}
\label{subsec:fps-qpoch}

Throughout this section $R$ is a commutative ring (with \texttt{DiscreteTopology}).

\begin{definition}[FPS finite $q$-Pochhammer symbol]\label{def:fps-qPoch}\lean{qSeries.FPS.qPoch}\leanok
For $a \in R[\![ X]\!]$ and $n \in \mathbb{N}$,
\[
  \texttt{qPoch}\,a\,n
  \;:=\; \prod_{k=0}^{n-1}\bigl(1 - a\cdot X^k\bigr)
  \;\in\; R[\![ X]\!].
\]
\end{definition}

\begin{lemma}[Base case]\label{lem:fps-qPoch-zero}\lean{qSeries.FPS.qPoch_zero}\leanok\uses{def:fps-qPoch}
$\texttt{qPoch}\,a\,0 = 1$.
\end{lemma}

\begin{proof}\leanok Empty product. \end{proof}

\begin{lemma}[FPS recurrence]\label{lem:fps-qPoch-succ}\lean{qSeries.FPS.qPoch_succ}\leanok\uses{def:fps-qPoch}
$\texttt{qPoch}\,a\,(n+1) = \texttt{qPoch}\,a\,n \cdot (1 - a\cdot X^n)$.
\end{lemma}

\begin{proof}\leanok Immediate from \texttt{Finset.prod\_range\_succ}. \end{proof}

\begin{lemma}[FPS shift identity]\label{lem:fps-qPoch-shift}\lean{qSeries.FPS.qPoch_eq_mul_qPoch_shift}\leanok\uses{def:fps-qPoch}
$\texttt{qPoch}\,a\,(n+1) = (1-a)\cdot\texttt{qPoch}\,(aX)\,n$.
\end{lemma}

\begin{proof}\leanok Induction on $n$ using the recurrence. \end{proof}

\begin{lemma}[Coefficient stabilisation]\label{lem:fps-coeff-stable}\lean{qSeries.FPS.coeff_qPoch_stable}\leanok\uses{def:fps-qPoch}
If $d < n$ then $[X^d]\,\texttt{qPoch}\,a\,(n+1) = [X^d]\,\texttt{qPoch}\,a\,n$.
\end{lemma}

\begin{proof}\leanok
The new factor $(1 - a X^n)$ only affects monomials of degree $\ge n > d$.
\end{proof}

\begin{lemma}[Constant coefficient]\label{lem:fps-constantCoeff-qPoch-X}\lean{qSeries.FPS.constantCoeff_qPoch_X}\leanok\uses{def:fps-qPoch}
The constant term of $\texttt{qPoch}\,(X : R[\![ X]\!])\,n$ is $1$.
\end{lemma}

\begin{proof}\leanok Induction on $n$; each factor $(1 - X \cdot X^k)$ has constant term $1$. \end{proof}

\begin{theorem}[FPS multipliability in pi topology]\label{thm:fps-multipliable}\lean{qSeries.FPS.multipliable_qPoch}\leanok\uses{def:fps-qPoch}
The family $k \mapsto 1 - a\cdot X^k$ is multipliable in $R[\![ X]\!]$
(with the pi / $X$-adic topology).
\end{theorem}

\begin{proof}\leanok
Apply \texttt{MvPowerSeries.WithPiTopology.multipliable\_one\_add\_of\_tendsto\_order\_atTop\_nhds\_top}:
the $X$-adic order of $-(a X^k)$ is $\ge k$ (the factor
$-(a X^k)$ is divisible by $X^k$), so the orders tend to $\infty$,
which is the criterion for multipliability in the pi topology.
No norm hypothesis is needed.
\end{proof}

\begin{definition}[FPS infinite $q$-Pochhammer symbol]\label{def:fps-qPochInf}\lean{qSeries.FPS.qPochInf}\leanok\uses{thm:fps-multipliable}
\[
  \texttt{qPochInf}\,a
  \;:=\; \prod_{k=0}^{\infty}\bigl(1 - a\cdot X^k\bigr)
  \;\in\; R[\![ X]\!].
\]
\end{definition}

\begin{lemma}[Coefficient formula]\label{lem:fps-coeff-qPochInf}\lean{qSeries.FPS.coeff_qPochInf}\leanok\uses{def:fps-qPochInf, lem:fps-coeff-stable}
$[X^d]\,\texttt{qPochInf}\,a = [X^d]\,\texttt{qPoch}\,a\,(d+1)$.
\end{lemma}

\begin{proof}\leanok
The partial products stabilise at degree $d$ once $n > d$
(Lemma~\ref{lem:fps-coeff-stable}), and converge to
$\texttt{qPochInf}\,a$ in the pi topology; continuity of
coefficient extraction gives the identity.
\end{proof}

\begin{lemma}[FPS recursion]\label{lem:fps-qPochInf-recursion}\lean{qSeries.FPS.qPochInf_recursion}\leanok\uses{def:fps-qPochInf}
$\texttt{qPochInf}\,a = (1-a)\cdot\texttt{qPochInf}\,(a\cdot X)$.
\end{lemma}

\begin{proof}\leanok
Extract the $k=0$ factor from the infinite product and shift:
$\prod_{k\ge 0}(1-a X^k) = (1-a)\cdot\prod_{k\ge 0}(1 - a X^{k+1})
= (1-a)\cdot\texttt{qPochInf}(aX)$.
\end{proof}

\begin{lemma}[Iterated telescoping]\label{lem:fps-qPochInf-eq-qPoch-mul}\lean{qSeries.FPS.qPochInf_eq_qPoch_mul}\leanok\uses{def:fps-qPochInf, def:fps-qPoch, lem:fps-qPochInf-recursion}
For every $n \in \mathbb{N}$:
$\texttt{qPochInf}\,a = \texttt{qPoch}\,a\,n \cdot \texttt{qPochInf}\,(a\cdot X^n)$.
\end{lemma}

\begin{proof}\leanok Induction on $n$ using Lemma~\ref{lem:fps-qPochInf-recursion}. \end{proof}

\begin{lemma}[FPS unit]\label{lem:fps-isUnit-qPochInf}\lean{qSeries.FPS.isUnit_qPochInf}\leanok\uses{def:fps-qPochInf}
If $1 - [X^0]\,a$ is a unit of $R$, then $\texttt{qPochInf}\,a$ is a unit
of $R[\![ X]\!]$.
\end{lemma}

\begin{proof}\leanok
The constant term of $\texttt{qPochInf}\,a$ equals $1 - [X^0]\,a$
(which is a unit), and a power series with unit constant term is a unit.
\end{proof}

\subsection{FPS Euler second identity}
\label{subsec:fps-euler2}

\begin{theorem}[FPS Euler second identity]\label{thm:fps-euler-second}\lean{qSeries.FPS.fps_euler_second}\leanok\uses{def:fps-qPochInf, def:fps-qPoch, def:qBinom, thm:qbin-finite, lem:fps-coeff-qPochInf}
In $R[\![ X]\!]$ (for any commutative ring $R$ with discrete topology),
\[
  \texttt{qPochInf}(-a)
  \;=\; \sum_{k=0}^{\infty} X^{\binom{k}{2}}\cdot a^k \cdot
        \bigl(\texttt{qPoch}\,(X : R[\![ X]\!])\,k\bigr)^{-1}.
\]
\end{theorem}

\begin{proof}\leanok
At each degree $d$, only finitely many terms contribute (those with
$\binom{k}{2} \le d$, i.e.\ $k \le d+1$). The key algebraic step is:
$[X^j]\,\texttt{qBinom}(N, k, X) = [X^j]\,(\texttt{qPoch}(X,k))^{-1}$
whenever $j + k \le N$ (\texttt{coeff\_qBinom\_eq\_qPochInv}).
Then the finite $q$-binomial theorem
(Theorem~\ref{thm:qbin-finite} with $z = -a$) gives
$\texttt{qPoch}(-a, N) = \sum_{k=0}^{N} X^{\binom{k}{2}} \texttt{qBinom}(N,k,X) (-a)^k$
in $R[\![ X]\!]$. Taking $N = d+1$ and extracting the degree-$d$
coefficient yields the identity at each degree. The sum is summable (each
degree has finitely many nonzero contributions) and the tsum specialises to
the tprod as $N \to \infty$ in the pi topology.
\end{proof}

\subsection{FPS key identity}
\label{subsec:fps-key}

\begin{definition}[FPS key sum]\label{def:fps-S}\lean{qSeries.FPS.S_fps}\leanok\uses{def:fps-qPochInf, def:fps-qPoch}
For $k \in \mathbb{N}$, define
\[
  S^{\mathrm{fps}}_k
  \;:=\; \sum_{m=0}^{\infty}
    X^{m(m+k)} \cdot \bigl(\texttt{qPoch}(X,m)\bigr)^{-1}
    \cdot \bigl(\texttt{qPoch}(X,m+k)\bigr)^{-1}
  \;\in\; R[\![ X]\!].
\]
\end{definition}

\begin{lemma}[FPS key sum is summable]\label{lem:fps-summable-S}\lean{qSeries.FPS.summable_S_fps}\leanok\uses{def:fps-S}
The summand of $S^{\mathrm{fps}}_k$ is summable in $R[\![ X]\!]$.
\end{lemma}

\begin{proof}\leanok
The monomial $X^{m(m+k)}$ has $X$-adic order $\ge m$; since $m \to \infty$, the
summand family satisfies the pi-topology criterion for summability.
\end{proof}

\begin{lemma}[FPS key sum recurrence]\label{lem:fps-S-recurrence}\lean{qSeries.FPS.S_fps_recurrence}\leanok\uses{def:fps-S, lem:fps-summable-S}
For every $k$:
\[
  S^{\mathrm{fps}}_k - S^{\mathrm{fps}}_{k+1}
  \;=\; X^{k+1}\cdot\bigl(S^{\mathrm{fps}}_{k+2} - S^{\mathrm{fps}}_{k+1}\bigr).
\]
\end{lemma}

\begin{proof}\leanok
Combine the two series term-by-term. Write
$(\texttt{qPoch}(X, m+k))^{-1} = (1 - X^{m+k+1})\cdot (\texttt{qPoch}(X, m+k+1))^{-1}$
(\texttt{qPochInv\_succ\_mul}), split the numerator, and reindex
the $(1-X^m)$-part by $m \mapsto m+1$.
This is a \emph{purely algebraic} ring identity in $R[\![ X]\!]$;
no norms are needed.
\end{proof}

\begin{theorem}[FPS: all $S_k$ are equal]\label{thm:fps-S-const}\lean{qSeries.FPS.S_fps_const}\leanok\uses{lem:fps-S-recurrence, lem:fps-summable-S}
$S^{\mathrm{fps}}_k = S^{\mathrm{fps}}_0$ for all $k \in \mathbb{N}$.
\end{theorem}

\begin{proof}\leanok
Let $D_k = S^{\mathrm{fps}}_k - S^{\mathrm{fps}}_{k+1}$.
By Lemma~\ref{lem:fps-S-recurrence}, $D_k = X^{k+1} D_{k+2} + X^{k+1} D_{k+1}$;
iterating, $[X^d]\,D_k = 0$ for all $d < k+1$ (directly from the recurrence,
since $[X^d]\,X^{k+1} = 0$ when $d < k+1$). For $d \ge k+1$, induction on $d$
using the recurrence (strong induction) yields $[X^d]\,D_k = 0$ as well,
because the only contributing term $[X^{k+1}]X^{k+1} = 1$ forces
$[X^{d-(k+1)}]\,D_{k+2} = [X^{d-(k+1)}]\,D_k$ and the argument closes
by strong induction. Hence $D_k = 0$ for all $k$, giving $S^{\mathrm{fps}}_k = S^{\mathrm{fps}}_0$.
\end{proof}

\begin{theorem}[FPS key identity]\label{thm:fps-key}\lean{qSeries.FPS.fps_key_identity}\leanok\uses{thm:fps-S-const, def:fps-S, def:fps-qPochInf}
For every $k$,
\[
  S^{\mathrm{fps}}_k
  \;=\; \bigl(\texttt{qPochInf}(X)\bigr)^{-1}
  \;=:\; \texttt{qqInv}.
\]
\end{theorem}

\begin{proof}\leanok
By Theorem~\ref{thm:fps-S-const}, $S^{\mathrm{fps}}_k = S^{\mathrm{fps}}_{d+1}$ for any $d$.
For $k = d+1 > d$, the only term contributing to $[X^d]\,S^{\mathrm{fps}}_{d+1}$
is $m=0$ (the $X^{0}$ monomial), giving $[X^d]\,(\texttt{qPoch}(X,d+1))^{-1}$.
For $d+1$ steps of the recursion \texttt{qPochInf}$= $\texttt{qPoch}$(X,n)\cdot$\texttt{qPochInf}$(X \cdot X^n)$,
one shows $[X^d]\,(\texttt{qPoch}(X,k))^{-1} = [X^d]\,\texttt{qqInv}$ for $d < k$
(Lemma \texttt{coeff\_qPochInv\_eq\_qqInv}). Hence $[X^d]\,S^{\mathrm{fps}}_k = [X^d]\,\texttt{qqInv}$ for all $d$.
\end{proof}

\paragraph{Comparison with the analytic key identity.}
The analytic key identity (Theorem~\ref{thm:S-sum-eq}) proves $S_k = 1/(q;q)_\infty$
using a \emph{norm estimate}: $\|D_k\| \le \|q\|^{n(2k+n+1)/2} \|D_{k+n}\| \to 0$.
The FPS key identity (Theorem~\ref{thm:fps-key}) proves the same identity
\emph{coefficient by coefficient} using only ring arithmetic in $R[\![ X ]\!]$,
requiring no analytic hypotheses whatsoever.

\subsection{FPS Cauchy coefficient identities}
\label{subsec:fps-cauchy}

\begin{lemma}[FPS product expansion]\label{lem:fps-qqInf-euler2}\lean{qSeries.FPS.qqInf_mul_euler2}\leanok\uses{def:fps-qPochInf, thm:fps-euler-second, lem:fps-qPochInf-eq-qPoch-mul}
In $R[\![ X]\!]$:
\[
  \texttt{qPochInf}(X)\cdot\texttt{qPochInf}(-a)
  \;=\; \sum_{n=0}^{\infty} X^{\binom{n}{2}}\,a^n\,
        \texttt{qPochInf}(X \cdot X^n).
\]
\end{lemma}

\begin{proof}\leanok
Expand $\texttt{qPochInf}(-a)$ via Theorem~\ref{thm:fps-euler-second} and distribute
$\texttt{qPochInf}(X)$ through the tsum. The identity
$\texttt{qPochInf}(X)\cdot(\texttt{qPoch}(X,n))^{-1} = \texttt{qPochInf}(X\cdot X^n)$
(Lemma~\ref{lem:fps-qPochInf-eq-qPoch-mul}) converts each term.
\end{proof}

\begin{theorem}[FPS Cauchy coefficient -- non-negative diagonal]\label{thm:fps-cauchy-nonneg}\lean{qSeries.FPS.cauchy_coeff_nonneg}\leanok\uses{lem:fps-qqInf-euler2, thm:fps-key, def:fps-S}
For every $k \ge 0$:
\[
  \sum_{m=0}^{\infty}
    \bigl[X^{(m+k)\text{.choose}\,2}\cdot\texttt{qPochInf}(X\cdot X^{m+k})\bigr]
    \cdot
    \bigl[X^{m\text{.choose}\,2}\cdot X^m \cdot (\texttt{qPoch}(X,m))^{-1}\bigr]
  \;=\; X^{k\text{.choose}\,2}.
\]
That is, the $k$-th diagonal sum of the Cauchy product equals $X^{\binom{k}{2}}$.
\end{theorem}

\begin{proof}\leanok
The diagonal sum equals $X^{\binom{k}{2}}\cdot\texttt{qPochInf}(X)\cdot S^{\mathrm{fps}}_k$
after applying the arithmetic identity
$\binom{m+k}{2} + \binom{m}{2} + m = \binom{k}{2} + m(m+k)$
(\texttt{choose2\_add}) and Lemma~\ref{lem:fps-qPochInf-eq-qPoch-mul}.
Apply Theorem~\ref{thm:fps-key} and $\texttt{qPochInf}(X)\cdot\texttt{qqInv} = 1$.
\end{proof}

\begin{theorem}[FPS Cauchy coefficient -- negative diagonal]\label{thm:fps-cauchy-neg}\lean{qSeries.FPS.cauchy_coeff_neg}\leanok\uses{lem:fps-qqInf-euler2, thm:fps-key, def:fps-S}
For every $l \ge 0$:
\[
  \sum_{n=0}^{\infty}
    \bigl[X^{n\text{.choose}\,2}\cdot\texttt{qPochInf}(X\cdot X^n)\bigr]
    \cdot
    \bigl[X^{(n+l+1)\text{.choose}\,2}\cdot X^{n+l+1} \cdot
          (\texttt{qPoch}(X,n+l+1))^{-1}\bigr]
  \;=\; X^{(l+2)\text{.choose}\,2}.
\]
\end{theorem}

\begin{proof}\leanok
Analogous to Theorem~\ref{thm:fps-cauchy-nonneg}, using the companion
arithmetic identity $\binom{n}{2} + \binom{n+l+1}{2} + (n+l+1) = \binom{l+2}{2} + n(n+l+1)$
(\texttt{choose2\_add'}) and $S^{\mathrm{fps}}_{l+1} = \texttt{qqInv}$.
\end{proof}

\subsection{FPS Jacobi triple product}
\label{subsec:fps-jtp}

\begin{definition}[FPS JTP objects]\label{def:fps-jtp}\lean{qSeries.FPS.jtpProd}\leanok\uses{def:fps-qPochInf}
Let $A = \texttt{LaurentPolynomial}\,\mathbb{C}$, $z = T(1)$,
$z^{-1} = T(-1)$ viewed as constant power series in $A[\![ X]\!]$.
Define:
\begin{align*}
  \texttt{jtpProd}
  &\;:=\; \texttt{qPochInf}(X)\cdot\texttt{qPochInf}(-z)\cdot\texttt{qPochInf}(-X\cdot z^{-1})
          \;\in\; A[\![ X]\!], \\
  \texttt{jtpSeries}
  &\;:=\; \Bigl(\sum_{n\ge 0} T(n)\cdot X^{n\text{.choose}\,2}\Bigr)
          + \Bigl(\sum_{m\ge 0} T(-(m+1))\cdot X^{(m+2)\text{.choose}\,2}\Bigr)
          \;\in\; A[\![ X]\!].
\end{align*}
Here $T(k)$ is the degree-$k$ monomial in the Laurent polynomial ring $A$,
so $T(n) = z^n$ and $T(-(m+1)) = z^{-(m+1)}$.
\end{definition}

\begin{theorem}[FPS Jacobi triple product]\label{thm:fps-jtp}\lean{qSeries.FPS.jacobiTripleProduct_fps}\leanok\uses{def:fps-jtp, thm:fps-cauchy-nonneg, thm:fps-cauchy-neg, lem:fps-qqInf-euler2, thm:fps-euler-second}
In $A[\![ X]\!]$ (with $A = \texttt{LaurentPolynomial}\,\mathbb{C}$):
\[
  \texttt{jtpProd} \;=\; \texttt{jtpSeries},
\]
that is,
\[
  \texttt{qPochInf}(X)\cdot\texttt{qPochInf}(-z)\cdot\texttt{qPochInf}(-X/z)
  \;=\;
  \sum_{k \ge 0} z^k X^{\binom{k}{2}}
  \;+\;
  \sum_{m \ge 0} z^{-(m+1)} X^{\binom{m+2}{2}}.
\]
\end{theorem}

\begin{proof}\leanok\proves{thm:fps-jtp}
Expand \texttt{jtpProd} using Lemma~\ref{lem:fps-qqInf-euler2} (on the
$\texttt{qPochInf}(X)\cdot\texttt{qPochInf}(-z)$ factor) and
Theorem~\ref{thm:fps-euler-second} (on $\texttt{qPochInf}(-X z^{-1})$), forming a
double tsum over $\mathbb{N}\times\mathbb{N}$. Summability of the double product
$a_n \cdot b_m$ (where $a_n = X^{\binom{n}{2}} z^n \texttt{qPochInf}(X\cdot X^n)$ and
$b_m = X^{\binom{m}{2}+m} z^{-m} (\texttt{qPoch}(X,m))^{-1}$) follows from the
pi-topology structure: each coefficient of $a_n \cdot b_m$ is eventually zero
in $n$ (once $\binom{n}{2} > d$) and in $m$ (once $\binom{m}{2}+m > d$).

Partition $\mathbb{N}\times\mathbb{N}$ via the bijection
$\mathtt{diagEquiv} : (n,m) \mapsto \text{inl}(n-m,m)$ if $m \le n$,
$\text{inr}(m-n-1,n)$ otherwise. The $k$-diagonal $(m \le n, k = n-m)$
sum equals $z^k X^{\binom{k}{2}}$ by Theorem~\ref{thm:fps-cauchy-nonneg};
the $l$-diagonal $(m > n, l = m-n-1)$ sum equals $z^{-(l+1)} X^{\binom{l+2}{2}}$
by Theorem~\ref{thm:fps-cauchy-neg}. Summing over all $k$ and $l$
gives \texttt{jtpSeries}.
\end{proof}

\paragraph{Comparison with Section~\ref{sec:implemented}.}
\begin{itemize}
\item The FPS proof (Theorem~\ref{thm:fps-jtp}) works over \emph{any} commutative
ring with discrete topology and needs no $\|q\|<1$ hypothesis; the analytic proof
(Theorem~\ref{thm:jtp-full}) works in $\mathbb{C}$ with $\|q\| < 1$ and $\|z\| < 1$.
\item The analytic proof is extended to all $z\neq 0$ by
Theorem~\ref{thm:jtp-analytic}; the FPS proof is already valid for all
$z \in A^\times$.
\item Both use the same algebraic infrastructure
(Sections~\ref{sec:qPochhammer-formalized}--\ref{subsec:qbin-finite-formalized}).
The key identity ($S_k = (q;q)_\infty^{-1}$) is proved analytically in
Theorem~\ref{thm:S-sum-eq} and algebraically (in $R[\![ X]\!]$) in
Theorem~\ref{thm:fps-key}.
\end{itemize}
